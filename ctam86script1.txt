-- create the product, sales agent, and customer tables
SHOW TABLES;
CREATE TABLE salesAgent (AgentID VARCHAR(2) NOT NULL, FirstName VARCHAR(30), LastName VARCHAR(30), WorkingCity VARCHAR(15), Commission INT, PRIMARY KEY (AgentID));

CREATE TABLE customer (CustomerID VARCHAR(2) NOT NULL, FirstName VARCHAR(30), LastName VARCHAR(30), City VARCHAR(15), PhoneNumber INT, PRIMARY KEY (CustomerID), FOREIGN KEY (AgentID) REFERENCES salesAgent(AgentID));
CREATE TABLE product (ProductID VARCHAR(2) NOT NULL, Description VARCHAR(20), CostPerItem INT, QuantityOnHand INT, PRIMARY KEY (ProductID), FOREIGN KEY (CustomerID) REFERENCES customer(CustomerID));
